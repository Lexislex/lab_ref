def analyze_blood_test(patient_results, sex, age):
    """–ê–Ω–∞–ª–∏–∑ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ –æ–±—â–µ–≥–æ –∞–Ω–∞–ª–∏–∑–∞ –∫—Ä–æ–≤–∏"""

    print(f"   –í –Ω–æ—Ä–º–µ: {normal_count}")
